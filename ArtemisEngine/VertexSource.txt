#version 330

layout(location = 0) in vec3 attrPosition; 
layout(location = 1) in vec3 attrColor;
layout(location = 2) in vec2 attrTexCoord;

out vec2 vTexCoord;

out vec3 normal;
out vec3 position;

uniform mat4 unifProjection;
uniform mat4 unifCamera;
uniform mat4 unifWorld;

void main()
{
	mat4 viewWorldTransform = unifProjection * unifWorld;
	mat4 normalTransform = transpose(inverse(viewWorldTransform));
	vec4 position4 = viewWorldTransform * vec4(attrPosition.x, attrPosition.y, attrPosition.z, 0.0);

	vec3 normal = (normalTransform * vec4(attrColor, 0.0)).xyz;
	
	normal = normalize(normal);

	vec3 position = position4.xyz;

	vTexCoord = attrTexCoord;

	gl_Position = unifProjection * position4;//unifCamera * unifWorld * vec4(attrPosition, 1.0f);
}
